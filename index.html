class PayrollController extends BaseController {
    protected $payrollModel;
    
    public function calculate($employeeId) {
        $gaji = $this->payrollModel->getGaji($employeeId);
        $bpjs = $this->calculateBPJS($gaji);
        $pph21 = $this->calculatePPH21($gaji);
        
        return $this->respond([
            'gaji_bersih' => $gaji - $bpjs - $pph21
        ]);
    }
}
